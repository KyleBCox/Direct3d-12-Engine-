124  DirectX::XMVECTORF32 up = { 0.0f, 1.0f, 0.0f, 0.0f };
125

127  m_camRotationMatrix = XMMatrixRotationRollPitchYaw(m_pitch, m_yaw, 0);
128  // Transforms the forward facing vector with the rotation matrix

130  // Normalises the vector which has just been rotated

133  const auto RotateYTempMatrix = XMMatrixRotationY(m_yaw);
134

136  const auto camRight = XMVector3TransformCoord(DefaultRight, RotateYTempMatrix);
137  // Rotates the \property DefaultForward vector with the yaw rotation matrix \var RotateYTempMatrix
138  const auto camForward = XMVector3TransformCoord(DefaultForward, RotateYTempMatrix);
139  // Rotates the \var up vector with the yaw rotation matrix \var RotateYTempMatrix
140  up.v = XMVector3TransformCoord(up, RotateYTempMatrix);
141 
142  // The float value to multiply the \property m_direction vector coords X and Z
143  const auto multiplyerXZ = -0.3f;
144  // The float value to multiply the \property m_direction vector coord Y
145  const auto multiplyerY = 0.3f;
146 
147  // Multiplies the \property m_direction vector with the multiplyers \var multiplyerXZ and \var multiplyerY
148  const auto x = XMVectorGetX(m_direction) * multiplyerXZ;
149  const auto z = XMVectorGetZ(m_direction) * multiplyerXZ;
150  const auto y = XMVectorGetY(m_direction) * multiplyerY;
151 
152  // Creates XMVECTOR from \var x and multiplies it by \var camRight to get the x movement of the camera
153  const auto xAxisMovement = XMVectorMultiply(XMVectorSet(x, x, x, x), camRight);
154  // Creates XMVECTOR from \var z and multiplies it by \var camForward to get the z movement of the camera
155  const auto zAxisMovement = XMVectorMultiply(XMVectorSet(z, z, z, z), camForward);
156  // Creates XMVECTOR from \var y and multiplies it by the \var up vector to get the y movement of the camera
157  const auto yAxisMovement = XMVectorMultiply(XMVectorSet(y, y, y, y), up);
158 
159  // Add the movement vectors (\var xAxisMovement \var zAxisMovement \var yAxisMovement) to the position (\var pos) of the camera
160  (*pos) = XMVectorAdd(*pos, xAxisMovement);
161  (*pos) = XMVectorAdd(*pos, zAxisMovement);
162  (*pos) = XMVectorAdd(*pos, yAxisMovement);
163 
164  // Creates the variable which will be passed to the XMStoreFloat3 function to store the final vector as a XMFLOAT3 structure
165  XMFLOAT3 fullNewPosFloat3;
166  // Creates the multiplyer / offset for the camera orbitting the player.
167  const auto camMultiplyer = 10;
168  // Orbits the camera around the player (research sin waves and cos waves to understand more)
169  XMStoreFloat3(&fullNewPosFloat3, XMVectorSet(sin(m_yaw) * camMultiplyer, sin(m_pitch) * camMultiplyer, cos(m_yaw) * camMultiplyer, 1) + (*pos));

171  float newYPos;

174  m_terrainCollider->GetNewYPos(fullNewPosFloat3, &newYPos);
175 
176  // sets the \var newYPos if it is above terrain and the camera is not (\property c_cameraGroundOffset is the constant offset of how much the camera should be above ground)
177  if (newYPos > fullNewPosFloat3.y - c_cameraGroundOffset) {
178  fullNewPosFloat3.y = newYPos + c_cameraGroundOffset;
179  m_isUnderGround = true;
180  }
181  else {
182  m_isUnderGround = false;
183  }
184 
185  // Puts the new pos (after orbitting and terrain collision) in to an XMVECTOR structure
186  auto newPos = XMLoadFloat3(&fullNewPosFloat3);
187  // Creates the View Matrix and adds it to the \class ModelViewProjectionManager helper class
188  m_mvpManager->SetMatrix(1, XMMatrixTranspose(XMMatrixLookAtRH(newPos, *pos, up.v)));

191 }
