125  *
126  * @param imageData the out parameter of data which is the image
127  * @param resourceDescription description of the resouce that is filled by this function
128  * @param filename path and filename of the texture
129  * @param bytesPerRow the number of bytes per row

131  */
132 int TextureLoader::LoadImageDataFromFile(BYTE** imageData, D3D12_RESOURCE_DESC& resourceDescription, LPCWSTR filename, int &bytesPerRow)
133 {
134  HRESULT hr;
135 
136  // we only need one instance of the imaging factory to create decoders and frames
137  IWICImagingFactory *wicFactory = NULL;
138 
139  // reset decoder, frame, and converter, since these will be different for each image we load
140  IWICBitmapDecoder *wicDecoder = NULL;
141  IWICBitmapFrameDecode *wicFrame = NULL;
142  IWICFormatConverter *wicConverter = NULL;
143 
144  bool imageConverted = false;
145 
146  if (wicFactory == NULL)
147  {
148  // Initialize the COM library
149  CoInitialize(nullptr);
150 
151  // create the WIC factory
152  hr = CoCreateInstance(
153  CLSID_WICImagingFactory,
154  NULL,
155  CLSCTX_INPROC_SERVER,
156  IID_PPV_ARGS(&wicFactory)
157  );
158  if (FAILED(hr)) return 0;
159 
160  hr = wicFactory->CreateFormatConverter(&wicConverter);
161  if (FAILED(hr)) return 0;
162  }
163 
164  // load a decoder for the image
165  hr = wicFactory->CreateDecoderFromFilename(
166  filename, // Image we want to load in
167  NULL, // This is a vendor ID, we do not prefer a specific one so set to null
168  GENERIC_READ, // We want to read from this file
169  WICDecodeMetadataCacheOnLoad, // We will cache the metadata right away, rather than when needed, which might be unknown
170  &wicDecoder // the wic decoder to be created
171  );
172  if (FAILED(hr)) return 0;
173 
174  // get image from decoder (this will decode the "frame")
175  hr = wicDecoder->GetFrame(0, &wicFrame);
176  if (FAILED(hr)) return 0;
177 
178  // get wic pixel format of image
179  WICPixelFormatGUID pixelFormat;
180  hr = wicFrame->GetPixelFormat(&pixelFormat);
181  if (FAILED(hr)) return 0;
182 
183  // get size of image
184  UINT textureWidth, textureHeight;
185  hr = wicFrame->GetSize(&textureWidth, &textureHeight);
186  if (FAILED(hr)) return 0;
187 
188  // we are not handling sRGB types in this tutorial, so if you need that support, you'll have to figure
189  // out how to implement the support yourself
190 
191  // convert wic pixel format to dxgi pixel format
192  DXGI_FORMAT dxgiFormat = GetDXGIFormatFromWICFormat(pixelFormat);
193 
194  // if the format of the image is not a supported dxgi format, try to convert it
195  if (dxgiFormat == DXGI_FORMAT_UNKNOWN)
196  {
197  // get a dxgi compatible wic format from the current image format
198  WICPixelFormatGUID convertToPixelFormat = GetConvertToWICFormat(pixelFormat);
199 
200  // return if no dxgi compatible format was found
201  if (convertToPixelFormat == GUID_WICPixelFormatDontCare) return 0;
202 
203  // set the dxgi format
204  dxgiFormat = GetDXGIFormatFromWICFormat(convertToPixelFormat);
205 
206  // make sure we can convert to the dxgi compatible format
207  BOOL canConvert = FALSE;
208  hr = wicConverter->CanConvert(pixelFormat, convertToPixelFormat, &canConvert);
209  if (FAILED(hr) || !canConvert) return 0;
210 
211  // do the conversion (wicConverter will contain the converted image)
212  hr = wicConverter->Initialize(wicFrame, convertToPixelFormat, WICBitmapDitherTypeErrorDiffusion, 0, 0, WICBitmapPaletteTypeCustom);
213  if (FAILED(hr)) return 0;
214 
215  // this is so we know to get the image data from the wicConverter (otherwise we will get from wicFrame)
216  imageConverted = true;
217  }
218 
219  int bitsPerPixel = GetDXGIFormatBitsPerPixel(dxgiFormat); // number of bits per pixel
220  bytesPerRow = (textureWidth * bitsPerPixel) / 8; // number of bytes in each row of the image data
221  int imageSize = bytesPerRow * textureHeight; // total image size in bytes
222 
223  // allocate enough memory for the raw image data, and set imageData to point to that memory
224  *imageData = (BYTE*)malloc(imageSize);
225 
226  // copy (decoded) raw image data into the newly allocated memory (imageData)
227  if (imageConverted)
228  {
229  // if image format needed to be converted, the wic converter will contain the converted image
230  hr = wicConverter->CopyPixels(0, bytesPerRow, imageSize, *imageData);
231  if (FAILED(hr)) return 0;
232  }
233  else
234  {
235  // no need to convert, just copy data from the wic frame
236  hr = wicFrame->CopyPixels(0, bytesPerRow, imageSize, *imageData);
237  if (FAILED(hr)) return 0;
238  }
239 
240  // now describe the texture with the information we have obtained from the image
241  resourceDescription = {};
242  resourceDescription.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
243  resourceDescription.Alignment = 0; // may be 0, 4KB, 64KB, or 4MB. 0 will let runtime decide between 64KB and 4MB (4MB for multi-sampled textures)
244  resourceDescription.Width = textureWidth; // width of the texture
245  resourceDescription.Height = textureHeight; // height of the texture
246  resourceDescription.DepthOrArraySize = 1; // if 3d image, depth of 3d image. Otherwise an array of 1D or 2D textures (we only have one image, so we set 1)
247  resourceDescription.MipLevels = 1; // Number of mipmaps. We are not generating mipmaps for this texture, so we have only one level
248  resourceDescription.Format = dxgiFormat; // This is the dxgi format of the image (format of the pixels)
249  resourceDescription.SampleDesc.Count = 1; // This is the number of samples per pixel, we just want 1 sample
250  resourceDescription.SampleDesc.Quality = 0; // The quality level of the samples. Higher is better quality, but worse performance
251  resourceDescription.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN; // The arrangement of the pixels. Setting to unknown lets the driver choose the most efficient one
252  resourceDescription.Flags = D3D12_RESOURCE_FLAG_NONE; // no flags
253 
254  // return the size of the image. remember to delete the image once your done with it (in this tutorial once its uploaded to the gpu)
255  return imageSize;
256 }
